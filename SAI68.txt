/* prog. to store 3 x 3 matrix values and print in matrix format   */
#include<stdio.h>
#include<conio.h>
void main()
{
  int a[3][3],i,j;
  clrscr();
  /* input part */
  printf("\nEnter  3 x 3 matrix values  ");
  for(i=0;i<3;i++)
   for(j=0;j<3;j++)
     scanf("%d",&a[i][j]);

 /* output part */
  printf("\nGiven Matrix  \n");
  for(i=0;i<3;i++)
  {
   for(j=0;j<3;j++)
      printf("%d  ",a[i][j]);
   printf("\n");
  }

  getch();
}
