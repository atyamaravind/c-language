/* function part */

void fact(int x)
{
	     int y=1;
	     while(x>0)
	     {  y=y*x;
		x--;
	     }
	     printf("\n Factorial = %d",y);
}

void factors(int x)
{
	     int y;
	     for(y=1;y<=x;y++)
	      if(x%y==0)
		printf("\n Factor = %d",y);

}

void table(int x)
{
	     int y;
	     for(y=1;y<=12;y++)
	      printf("\n %d x %d = %d",x,y,x*y);
}

void prime(int x)
{
	int c,y;
	     c=0;
	     for(y=1;y<=x;y++)
	       if(x%y==0)
		 c++;
	     if(c==2)
	       printf("\n Prime");
	     else
	       printf("\nNot prime");
}
