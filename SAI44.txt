/* prog. to print GCD,LCM of two numbers using while  */
#include<stdio.h>
#include<conio.h>
void main()
{
  int a,b,gcd,r=1,lcm,x,y;
  clrscr();

  printf("\n Enter two numbers : ");
  scanf("%d%d",&a,&b);
  x=a;
  y=b;

  while(r!=0)
  {
    r=a%b;
    if(r==0)
      gcd=b;
    else
    {
      a=b;
      b=r;
    }
  }
 lcm=x*y/gcd;
 printf("\n GCD = %d    LCM = %d",gcd,lcm);
 getch();
}
