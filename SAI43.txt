/* prog. to print GCD,LCM of two numbers */
#include<stdio.h>
#include<conio.h>
void main()
{
  int a,b,gcd,r,lcm,x,y;
  clrscr();

  printf("\n Enter two numbers : ");
  scanf("%d%d",&a,&b);
  x=a;
  y=b;
  do
  {
    r=a%b;
    if(r==0)
      gcd=b;
    else
    {
      a=b;
      b=r;
    }
  }while(r!=0);
 lcm=x*y/gcd;
 printf("\n GCD = %d    LCM = %d",gcd,lcm);
 getch();
}
