#include<stdio.h>

void main()
{
  int ch,n,f,c=0;

  do
  {
  clrscr();
  printf("\n MULTIPLE OPERATIONS ");

  printf("\n 1. Factorial ");
  printf("\n 2. Factors ");
  printf("\n 3. Multiplication Table");
  printf("\n 4. Prime or not");
  printf("\n 5. Exit");
  printf("\n --------------------");


   printf("\nEnter your choice : ");
   scanf("%d",&ch);

   if(ch>0 && ch<5)
   {
    printf("\n Enter a number : ");
    scanf("%d",&n);
   }
  switch(ch)
  {
     case 1:
	     f=1;
	     while(n>0)
	     {  f=f*n;
		n--;
	     }
	     printf("\n Factorial = %d",f);
	     break;
     case 2:
	     for(f=1;f<=n;f++)

	       if(n%f==0)
		 c++;
	     if(c==2)
	       printf("\n Prime");
	     else
	       printf("\nNot prime");
	     break;

     case 5:
	      printf("\n Thank you for using this prog.");
	      break;
     default:
	      printf("\n Enter correct choice  ");

  }

}while(ch!=5);

}
