/* prog. to print factors,sum of factors, no. of factors of a number */
#include<stdio.h>
#include<conio.h>
void main()
{
  int n,i=1,r,c=0,sum=0;
  clrscr();
  printf("\n Enter a number : ");
  scanf("%d",&n);
  while(i<=n)
  {
    r=n%i;
    if(r==0)
    {
      printf("\n Factor = %d",i);
      sum=sum+i;
      c++;
    }
    i++;
  }
  printf("\n Sum = %d   No. of Factors = %d",sum,c);
 getch();
}
